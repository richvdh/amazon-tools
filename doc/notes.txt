Bootstrapping a backup instance
----
use ./bin/bootstrap-backup-snapshot to create snapshot

# on host:
apt-get install lockfile-progs rdiff-backup cstream

adduser --system --disabled-password amazon
echo "--region=eu-west-1" > ~amazon/.awsrc

# create IAM user and access key
# creds go in ~amazon/.awssecret, which should be chmod 600

cd /usr/local

git clone git@github.com:richvdh/amazon-tools.git
git submodule init
git submodule update

mkdir ~/backup
echo <snapid> > ~/backup/snapid

# config in /etc/backup
# wrapper script in /etc/cron.daily
